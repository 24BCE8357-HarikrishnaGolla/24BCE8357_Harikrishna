
SELECT W.FIRST_NAME, W.LAST_NAME, B.BONUS_AMOUNT
FROM Worker W
INNER JOIN Bonus B
ON W.WORKER_ID = B.WORKER_REF_ID;


SELECT W.FIRST_NAME, W.DEPARTMENT, B.BONUS_AMOUNT
FROM Worker W
LEFT JOIN Bonus B
ON W.WORKER_ID = B.WORKER_REF_ID;

SELECT W.FIRST_NAME, B.BONUS_AMOUNT
FROM Worker W
RIGHT JOIN Bonus B
ON W.WORKER_ID = B.WORKER_REF_ID;

SELECT W.FIRST_NAME, W.LAST_NAME, T.WORKER_TITLE
FROM Worker W
INNER JOIN Title T
ON W.WORKER_ID = T.WORKER_REF_ID;

SELECT 
    W.FIRST_NAME,
    W.LAST_NAME,
    T.WORKER_TITLE,
    B.BONUS_AMOUNT
FROM Worker W
INNER JOIN Title T
ON W.WORKER_ID = T.WORKER_REF_ID
INNER JOIN Bonus B
ON W.WORKER_ID = B.WORKER_REF_ID;

SELECT 
    A.FIRST_NAME AS EMPLOYEE1,
    B.FIRST_NAME AS EMPLOYEE2,
    A.DEPARTMENT
FROM Worker A
JOIN Worker B
ON A.DEPARTMENT = B.DEPARTMENT
AND A.WORKER_ID <> B.WORKER_ID;
